# Recruitment Queue System Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace auto-close with a queue system, show participants vertically in the post, add duplicate-join/leave validation, and add a manual reopen button.

**Architecture:** DB schema change removes `autoClose`/`autoClosed` and adds `isQueued`/`queuedAt` to participants. All interaction handlers are updated to check participant state before acting. The post message now renders active participants and queue members as vertical mention lists.

**Tech Stack:** TypeScript, Discord.js (Components V2), Drizzle ORM (PostgreSQL), Node.js test runner

export async function joinRecruitment(
  db: DbClient,
  input: { recruitmentId: string; userId: string; isQueued?: boolean; joinedAt?: Date; queuedAt?: Date | null }
) {
  const isQueued = input.isQueued ?? false;
  const queuedAt = isQueued ? (input.queuedAt ?? new Date()) : null;

  const [participant] = await db
    .insert(recruitmentParticipants)
    .values({

      userId: input.userId,
      joinedAt: input.joinedAt ?? new Date(),

        updatedAt: sql`now()`

```ts
export async function promoteFromQueue(
  db: DbClient,
  recruitmentId: string
) {
  const [first] = await db
    .select()
    .from(recruitmentParticipants)
    .where(
      and(
        eq(recruitmentParticipants.recruitmentId, recruitmentId),
        isNull(recruitmentParticipants.leftAt),
        eq(recruitmentParticipants.isQueued, true)
      )
    )
    .orderBy(asc(recruitmentParticipants.queuedAt))
    .limit(1);

  if (!first) return null;

  await db
    .update(recruitmentParticipants)
    .set({ isQueued: false, queuedAt: null, updatedAt: sql`now()` })
    .where(eq(recruitmentParticipants.id, first.id));

  return { userId: first.userId };
}
```

export function createRecruitmentPostMessage(
  recruitment: Recruitment,
  loc: Loc,
  activeParticipantCount = 0,
  participantIds: string[] = [],

): MessageCreateOptions & MessageEditOptions {
  const isClosed = recruitment.status === "closed";
  const vcText = recruitment.voiceChannelId
    ? loc.recruitmentPostVc({ id: recruitment.voiceChannelId })
    : loc.recruitmentPostNoVc;

  const participantText =
    participantIds.length > 0

        customId: createRecruitmentCustomId("reopen", recruitment.id),
        label: loc.recruitmentButtonReopen,
        style: ButtonStyle.Success
      }
    : {
        type: ComponentType.Button as const,
        customId: createRecruitmentCustomId("close", recruitment.id),
        label: loc.recruitmentButtonClose,
        style: ButtonStyle.Danger
      };

  const containerComponents = [

      type: ComponentType.TextDisplay,
      content: `## ${loc.recruitmentPostTitle({ title: recruitment.genre })}`
    },
    {
      type: ComponentType.TextDisplay,
      content: `${localizeStatus(recruitment.status, loc)}  ·  👥 ${activeParticipantCount} / ${recruitment.capacity}`
    },
    { type: ComponentType.Separator },

      content: recruitment.content
    },
    { type: ComponentType.Separator },
    {
      type: ComponentType.TextDisplay,
      content: loc.recruitmentPostCreator({ id: recruitment.creatorId })
    },
    {
      type: ComponentType.TextDisplay,
      content: vcText
    },
    {
      type: ComponentType.TextDisplay,
      content: participantText
    },
    ...(queueText
      ? [{ type: ComponentType.TextDisplay, content: queueText }]
      : [])
  ];

  return {
    flags: MessageFlags.IsComponentsV2,
    components: [
      {
        type: ComponentType.Container,
        accent_color: EVENT_COLORS.teal,

            style: ButtonStyle.Secondary

 * DB call order:
 * 1. resolveLocale → getGuildConfigByGuildId (innerJoin) → []
 * 2. getRecruitmentById → [recruitment]
 * 3. getActiveParticipant → [existing] or []
 * 4. countActiveRecruitmentParticipants → [{value: activeCount}]
 * 5. joinRecruitment → [participant]
 * 6. updateRecruitmentStatus (if status changed) → [updated]
 * 7. listActiveRecruitmentParticipants → participants[]
 * 8. listQueuedParticipants → queued[]
 */
function makeJoinDb(opts: {
  recruitment: ReturnType<typeof makeRecruitment>;
  existing?: { isQueued: boolean } | null;
  activeCount: number;
  activeParticipants?: { userId: string }[];
  queuedParticipants?: { userId: string }[];
  updatedRecruitment?: ReturnType<typeof makeRecruitment>;
}) {
  const existing = opts.existing ?? null;
  const activeParticipants = opts.activeParticipants ?? [];
  const queuedParticipants = opts.queuedParticipants ?? [];
  const updatedRecruitment = opts.updatedRecruitment ?? opts.recruitment;

  let selectCallCount = 0;

  return {
    select: (selection?: Record<string, unknown>) => {
      const isGuildConfig = selection && "language" in selection;
      const isCount = selection && "value" in selection;
      selectCallCount++;

      if (isGuildConfig) {
        return { from: () => ({ innerJoin: () => ({ where: () => ({ limit: async () => [] }) }) }) };
      }
      if (isCount) {
        return { from: () => ({ where: async () => [{ value: opts.activeCount }] }) };
      }

      // selectCallCount 1 = getRecruitmentById, 2 = getActiveParticipant,
      // 3 = listActiveRecruitmentParticipants, 4 = listQueuedParticipants
      if (selectCallCount === 2) {
        // getActiveParticipant: returns limit(1)
        return {
          from: () => ({
            where: () => ({
              limit: async () => existing ? [{ ...existing, userId: "user-1", recruitmentId: opts.recruitment.id, leftAt: null }] : []
            })
          })
        };
      }
      if (selectCallCount === 3) {
        // listActiveRecruitmentParticipants: returns array (orderBy)
        return { from: () => ({ where: () => ({ orderBy: async () => activeParticipants }) }) };
      }
      if (selectCallCount === 4) {
        // listQueuedParticipants
        return { from: () => ({ where: () => ({ orderBy: async () => queuedParticipants }) }) };
      }
      // Default: getRecruitmentById
      return { from: () => ({ where: () => ({ limit: async () => [opts.recruitment] }) }) };
    },
    insert: () => ({
      values: () => ({
        onConflictDoUpdate: () => ({
          returning: async () => [{ userId: "user-1", recruitmentId: opts.recruitment.id, isQueued: false, leftAt: null }]
        })
      })
    }),
    update: () => ({
      set: () => ({ where: () => ({ returning: async () => [updatedRecruitment] }) })
    })
  };
}
